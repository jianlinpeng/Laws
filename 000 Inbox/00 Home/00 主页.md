```dataview
TABLE WITHOUT ID
file.link AS "文件名",
dateformat(发布日期, "yyyy-MM-dd") AS "发布日期",
dateformat(施行日期, "yyyy-MM-dd") AS "施行日期",
choice(施行日期 > date(now), "⏳ 尚未生效", "✅ 已生效") AS 时效性
WHERE 施行日期 AND 施行日期 > date("2025-01-01")
SORT 施行日期 DESC
LIMIT 100
```
